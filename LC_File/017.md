#### 旋转数组

---

题目描述:  给定一个数组，将数组中的元素向右移动 *k* 个位置，其中 *k* 是非负数

**思路**

```python
class Solution:
    # 更费时一些，主要在list.insert()是o(n)
    def rotate(self, nums, k):
        """
        Do not return anything, modify nums in-place instead.
        """
        if k>=len(nums):
            k -= len(nums)
        while k:
            value = nums.pop()
            nums.insert(0,value)
            k -= 1
    # 空间换时间
    def rotateII(self,nums,k):
        if k>=len(nums):
            k -= len(nums)
        q,h = [],[]
        while nums:
            if k:
                q.append(nums.pop())
                k -= 1
            else:
                h.append(nums.pop())
        for i in reversed(q):
            nums.append(i)
        for i in reversed(h):
            nums.append(i)
        print(nums)

s = Solution()
nums = [1,2,3,4,5,6,7]
k = 3
s.rotateII(nums,k)
print(nums)
        
```
